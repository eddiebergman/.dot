%%%%%%%%%%%%%%%%%%%%
% Style
%%%%%%%%%%%%%%%%%%%%
\mdfdefinestyle{mdProposition}{%
	skipabove=8pt,
	linewidth=0.6pt,
	rightline=false, leftline=false, topline=false, bottomline=false,
	linecolor=\cPropositionLine
}


\declaretheoremstyle[
	spaceabove=12pt,
	spacebelow=5pt,
	headfont={\fPropositionHead},
	notefont={\fPropositionNote},
	bodyfont={\fPropositionBody},
	headpunct={},
	postheadspace=\newline
]{thmProposition}

\declaretheorem[
    name=Proposition, sibling=chaptercounter, style=thmProposition
]{propositiontheorem}

%%%%%%%%%%%%%%%%%%%%
% Env: Proposition
%%%%%%%%%%%%%%%%%%%%
% [] #1: pgfkeys [
%       Bool numbered      % Numbers the Subsection of the proposition
%       Bool newpage       % Creates a blank page after the Subsection
%   ]
% {} #2: definition name
\newenvironment{proposition_env}[2][]
{
	\begin{subsection_env}[#1]{#2}
    \begin{mdframed}[style=mdProposition]
	\begin{propositiontheorem}
	\label{proposition:#2}
}
{
    \end{propositiontheorem}
	\end{mdframed}
	\end{subsection_env}
}
