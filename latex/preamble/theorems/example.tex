\mdfdefinestyle{mdExample}{
	skipabove=8pt,
	linewidth=0.5pt,
	rightline=false, leftline=false, topline=true, bottomline=false,
	linecolor=\cExampleLine,
}

\declaretheoremstyle[
	headfont=\bfseries\color{\cExampleHeadFont},
	headpunct={\\[3pt]},
	postheadspace={0pt},
	spaceabove=12pt,
	spacebelow=8pt
]{thmExample}


\declaretheorem[
    name=Example, numberwithin=chapter, style=thmExample
]{exampletheorem}

\newenvironment{example_env}
{
    \begin{mdframed}[style=mdExample]
    \begin{exampletheorem}
}
{
    \end{exampletheorem}
    \end{mdframed}
}
