%%%%%%%%%%%%%%%%%%%%
% Style
%%%%%%%%%%%%%%%%%%%%
\mdfdefinestyle{mdDefinition}{%
	skipabove=8pt,
	linewidth=0.6pt,
	rightline=false, leftline=false, topline=false, bottomline=false,
	linecolor=\cDefinitionLine
}

\declaretheoremstyle[
	headfont={\fDefinitionHead},
	notefont={\fDefinitionNote},
	bodyfont={\fDefinitionBody},
	headpunct={},
	spaceabove=12pt,
	spacebelow=8pt,
	postheadspace=\newline,
	% postheadhook={\textcolor{gray!80}{ \rule[.8ex]{2em}{0.4pt} }}
]{thmDefinition}

\declaretheorem[
    name=Definition, sibling=chaptercounter, style=thmDefinition
]{definitiontheorem}

%%%%%%%%%%%%%%%%%%%%
% Env: Definition
%%%%%%%%%%%%%%%%%%%%
% [] #1: pgfkeys [
%       Bool numbered      % Numbers the Subsection of the definition
%       Bool newpage       % Creates a blank page after the Subsection
%   ]
% {} #2: definition name
\newenvironment{definition_env}[2][]
{
	\begin{subsection_env}[#1]{#2}
    \begin{mdframed}[style=mdDefinition]
    \begin{definitiontheorem}
	\label{definition:#2}
}
{
    \end{definitiontheorem}
	\end{mdframed}
	\end{subsection_env}
}
