%%%%%%%%%%%%%%%%%%%%
% Style
%%%%%%%%%%%%%%%%%%%%
\mdfdefinestyle{mdTheorem}{%
	skipabove=8pt,
	linewidth=0.5pt,
	rightline=false, leftline=false, topline=true, bottomline=false,
	linecolor=\cTheoremLine
}

\declaretheoremstyle[
	headfont=\fTheoremHead,
	notefont=\fTheoremNote,
	bodyfont=\fTheoremBody,
	headpunct={\\[3pt]},
	postheadspace={0pt},
	spaceabove=12pt,
	spacebelow=8pt
]{thmTheorem}

\declaretheorem[
    name=Theorem, numberwithin=chaptercounter, style=thmTheorem
]{theoremtheorem}

%%%%%%%%%%%%%%%%%%%%
% Env: Theorem
%%%%%%%%%%%%%%%%%%%%
% [] #1: pgfkeys [
%       Bool numbered      % Numbers the Subsection of the theorem
%       Bool newpage       % Creates a blank page after the Subsection
%   ]
% {} #2: theorem name
\newenvironment{theorem_env}[2][]
{
	\begin{subsection_env}[#1]{#2}
    \begin{mdframed}[style=mdTheorem]
	\begin{theoremtheorem}
	\label{theorem:#2}
}
{
    \end{theoremtheorem}
    \end{mdframed}
}
